<script>

let items = [
  {name: "Pizza", img: "https://source.unsplash.com/400x400/?pizza"},
  {name: "Sushi", img: "https://source.unsplash.com/400x400/?sushi"},
  {name: "Mare", img: "https://source.unsplash.com/400x400/?sea"},
  {name: "Montagna", img: "https://source.unsplash.com/400x400/?mountain"},
  {name: "Cinema", img: "https://source.unsplash.com/400x400/?cinema"},
  {name: "Videogiochi", img: "https://source.unsplash.com/400x400/?gaming"},
  {name: "Estate", img: "https://source.unsplash.com/400x400/?summer"},
  {name: "Inverno", img: "https://source.unsplash.com/400x400/?winter"}
];

let currentRound = [];
let winners = [];
let semifinalLosers = [];
let secondPlace;

function shuffle(array) {
  return array.sort(() => Math.random() - 0.5);
}

function startTournament() {
  currentRound = shuffle([...items]);
  nextMatch();
}

function nextMatch() {
  if (currentRound.length === 1) {
    showPodium(currentRound[0]);
    return;
  }

  if (currentRound.length === 2) {
    semifinalLosers.push(null);
  }

  let a = currentRound.shift();
  let b = currentRound.shift();

  if (!b) {
    winners.push(a);
    prepareNextRound();
    return;
  }

  document.getElementById("game").innerHTML = `
    <div class="container">
      <div class="card" onclick="chooseWinner('${a.name}')">
        <img src="${a.img}">
        <h3>${a.name}</h3>
      </div>
      <div class="card" onclick="chooseWinner('${b.name}')">
        <img src="${b.img}">
        <h3>${b.name}</h3>
      </div>
    </div>
  `;

  window.currentPair = {a, b};
}

function chooseWinner(name) {
  let winner = window.currentPair.a.name === name ? window.currentPair.a : window.currentPair.b;
  let loser = window.currentPair.a.name === name ? window.currentPair.b : window.currentPair.a;

  if (currentRound.length === 0 && winners.length === 0) {
    secondPlace = loser;
  }

  if (currentRound.length === 0 && winners.length === 1) {
    secondPlace = loser;
  }

  if (currentRound.length === 2) {
    semifinalLosers.push(loser);
  }

  winners.push(winner);
  nextMatch();
}

function prepareNextRound() {
  currentRound = winners;
  winners = [];
  nextMatch();
}

function showPodium(first) {
  let third = semifinalLosers[0] || null;

  document.getElementById("game").innerHTML = "";
  document.getElementById("result").innerHTML = `
    <h2>🏆 Vincitore: ${first.name}</h2>
    <h3>🥈 Secondo: ${secondPlace ? secondPlace.name : ""}</h3>
    <h3>🥉 Terzo: ${third ? third.name : ""}</h3>
    <br>
    <button onclick="location.reload()">Rigioca</button>
  `;
}

startTournament();
